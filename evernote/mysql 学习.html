mysql 命令行学习：
1.显示数据库
         SHOW DATABASES; (mysql 命令不区分大小写)    
2.创建数据库
        create database test;
3.打开数据库
        use test;
4.显示所有表格
        show tables;
5.创建表格
        CREATE TABLE Persons( primary_id INT AUTO_INCREMENT PRIMARY KEY, LastName varchar(255),FirstName varchar(255),Address varchar(255))
6.删除数据库|删除表格
      DROP DATABASE 数据库名称；
    DROP TABLE 表名称；
7.清空表
    truncate table 表名称；
8.查看数据库所存放物理地址、
     show global variables like "%datadir%";
9.添加主键
        alter table lzw_test add primary key(name);
         alter table lzw_test drop primary key;
10.添加索引与删除索引
        alter table lzw_test add index name_index(name);
        alter table lzw_test drop index name_index;
11.表重命名
         alter table name rename group_member;
