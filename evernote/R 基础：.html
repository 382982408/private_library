8）数据结构 （标向阵组框因列）
    标量     ->    只含一个元素的向量        g  <-    3
    向量     ->    必须相同数据类型            g  <-    c(3,4)
    矩阵    ->    必须相同数据类型，2维        g  <-   matrix(1:20, nrow=5, ncol=4, byrow=TRUE, dimnames=list(c("name1","name2"...),c("name1"...)))
    数组    ->    必须相同数据类型，3维甚至多维    g  <-  array(1:24,c(2,3,4), dimnames=list(c("name1","name2"...),c("name1"...),c("name1"...)))
    数据框    ->    可以不同数据类型，3维甚至多维  g <- data.frame(vector_1, vector_2, vector_3, vector_4)
    因子     ->    g<- factor(c('green','green','yellow','red','red','red','green')),  
    列表    ->   2维，里面存放各种不同类型。

            cex.lab         坐标轴大小
            cex.main        标题大小
            cex.sub            副标题大小
    lty          ->        线条类型 （1.。。6）
    lwd        ->        线条宽度（实数，默认大小的倍数）
    col         ->        绘图颜色
            col.axis        坐标轴刻度文字颜色
            col.lab         坐标轴颜色
            col.main        标题颜色
            col.sub            副标题颜色
            fq                图形前景色

    font    ->        字体样式，1=常规，2=粗体，3=斜体，4=粗斜体，5=符号字体。。。。
           font.axis        坐标轴刻度字体样式
           font.lab         坐标轴标签字体样式

           font.sub            副标题字体样式
    legend （location,title,legend...）
            location    位置
            titile          图例标题字符串
            legend       图例标签组成的字符串向量
    xlab="xxx"
    ylab="yyy"

    ylim=c(0,40)
    也可使用title 函数来添加标题

    坐标轴函数axis
            axis(side,at=,labels=,pos=,lty=,col=,las=,tck=,)
                side    一个整数，1,2,3,4 分别表示下左上右
                at        数值型向量，表示需要绘制刻度线的位置
                labels    字符型向量，刻度旁文字标签
                pos        坐标轴相交位置坐标
                las        0表示平行，2表示垂直
                tck        0~1；0表示禁用刻度，1表示绘制网格线
