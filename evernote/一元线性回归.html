#!/usr/bin/env python
import sys,os,re,getopt,pandas,numpy
import matplotlib.pyplot
from sklearn import linear_model
from io import StringIO
import matplotlib.pyplot as plt
####################################################
if len(sys.argv)<2:
       print ("\n\tpython %s <input_file>\n"%(sys.argv[0]));
       sys.exit(0);
else:
       print ("work start...\n");
####################################################
data_keys=[];
data_values=[];
data_keys22=[];
print ("open file %s ...\n" % sys.argv[1]);
IN=open(sys.argv[1],"r");
for LINE in IN:
       LINE=re.sub("\n","",LINE);
       (key,value)=LINE.split("\t");
       key=float(key);
       value=float(value);
       data_keys.append(key);
       data_values.append(value);
       key_list=[];
       key_list.append(key);
       data_keys22.append(key_list);
IN.close();
#print (data_keys);
#print (data_values);
regr = linear_model.LinearRegression();
regr.fit(numpy.array(data_keys22),numpy.array(data_values));
a,b=regr.coef_,regr.intercept_;
print (("a= %f \nb= %f \n") % (a,b));
print ("work end ...\n");
print ([data_keys]);
print ([data_values]);
print (data_keys22);
plt.scatter(numpy.array(data_keys),numpy.array(data_values),color='blue')
#plt.scatter([data_keys],[data_values],color='blue')
#print(numpy.array(data_keys),regr.predict(data_keys22),[data_keys])
plt.plot(numpy.array(data_keys),regr.predict(numpy.array(data_keys22)), color='red', linewidth=4)
plt.show()
